using System;
using CurrencyConverterAPI.Services;
using Microsoft.AspNetCore.Http;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Logging;
using Moq;
using Xunit;

namespace CurrencyConverterAPI.Tests.Services;

public class HeaderBasedCurrencyConverterFactoryTests
{
    private readonly Mock<IHttpContextAccessor> _mockHttpContextAccessor;
    private readonly Mock<ILogger<HeaderBasedCurrencyConverterFactory>> _mockLogger;
    private readonly IServiceProvider _serviceProvider;
    private readonly HeaderBasedCurrencyConverterFactory _sut;
    private readonly Mock<ICurrencyConverter> _mockDefaultConverter;
    private readonly Mock<ICurrencyConverter> _mockTestProviderConverter;

    public HeaderBasedCurrencyConverterFactoryTests()
    {
        _mockHttpContextAccessor = new Mock<IHttpContextAccessor>();
        _mockLogger = new Mock<ILogger<HeaderBasedCurrencyConverterFactory>>();

        _mockDefaultConverter = new Mock<ICurrencyConverter>();
        _mockDefaultConverter.Setup(c => c.ProviderName).Returns("FrankfurterApiCurrencyConverter");

        _mockTestProviderConverter = new Mock<ICurrencyConverter>();
        _mockTestProviderConverter.Setup(c => c.ProviderName).Returns("TestProvider");

        var services = new ServiceCollection();
        services.AddKeyedSingleton("FrankfurterApiCurrencyConverter", _mockDefaultConverter.Object);
        services.AddKeyedSingleton("TestProvider", _mockTestProviderConverter.Object);
        _serviceProvider = services.BuildServiceProvider();

        _sut = new HeaderBasedCurrencyConverterFactory(
            _mockHttpContextAccessor.Object,
            _mockLogger.Object,
            _serviceProvider);
    }

    [Fact]
    public void ResolveService_ShouldReturnProviderFromHeader_WhenHeaderIsValid()
    {
        // Arrange
        const string providerName = "TestProvider";
        var httpContext = new DefaultHttpContext();
        httpContext.Request.Headers["X-Currency-Provider"] = providerName;
        _mockHttpContextAccessor.Setup(x => x.HttpContext).Returns(httpContext);

        // Act
        var result = _sut.ResolveService();

        // Assert
        Assert.NotNull(result);
        Assert.Equal(providerName, result.ProviderName);
    }

    [Theory]
    [InlineData(null)] // Header not present
    [InlineData("")]   // Header present but empty
    [InlineData(" ")]  // Header present but whitespace
    public void ResolveService_ShouldReturnDefaultProvider_WhenHeaderIsMissingOrEmpty(string headerValue)
    {
        // Arrange
        var httpContext = new DefaultHttpContext();
        if (headerValue is not null)
        {
            httpContext.Request.Headers["X-Currency-Provider"] = headerValue;
        }
        _mockHttpContextAccessor.Setup(x => x.HttpContext).Returns(httpContext);

        // Act
        var result = _sut.ResolveService();

        // Assert
        Assert.NotNull(result);
        Assert.Equal("FrankfurterApiCurrencyConverter", result.ProviderName);
    }

    [Fact]
    public void ResolveService_ShouldReturnDefaultProvider_WhenProviderInHeaderIsUnsupported()
    {
        // Arrange
        const string providerName = "UnsupportedProvider";
        var httpContext = new DefaultHttpContext();
        httpContext.Request.Headers["X-Currency-Provider"] = providerName;
        _mockHttpContextAccessor.Setup(x => x.HttpContext).Returns(httpContext);

        // Act
        var result = _sut.ResolveService();

        // Assert
        Assert.NotNull(result);
        Assert.Equal("FrankfurterApiCurrencyConverter", result.ProviderName);
    }

    [Fact]
    public void ResolveService_ShouldReturnDefaultProvider_WhenHttpContextIsNull()
    {
        // Arrange
        _mockHttpContextAccessor.Setup(x => x.HttpContext).Returns((HttpContext)null);

        // Act
        var result = _sut.ResolveService();

        // Assert
        Assert.NotNull(result);
        Assert.Equal("FrankfurterApiCurrencyConverter", result.ProviderName);
    }
}
